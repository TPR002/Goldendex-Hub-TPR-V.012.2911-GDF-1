GoldendexLauncher.GoldendexLauncher.Enabled = false;

local launcherData = game:GetService("HttpService"):JSONDecode(
	(game:HttpGet("https://raw.githubusercontent.com/TPR002/Goldendex-Hub-TPR-V.012.2911-GDF-1/refs/heads/main/Goldendex%20Hub%20Admin2.txt")));

GoldendexLauncher.GoldendexLauncher.Enabled = true;

-- Startup animation

-- Essentially changes the visibillity of the descendants of the main frame of the launcher
local function changeElementState(state)
	for i,v in pairs(GoldendexLauncher.GoldendexLauncher.Frame:GetDescendants()) do
		if v:IsA("GuiObject") then
			v.Visible = state;
		end
	end
end

changeElementState(false);

local oldSize = GoldendexLauncher.GoldendexLauncher.Frame.Size;
GoldendexLauncher.GoldendexLauncher.Frame.Size = UDim2.new(0,0,0,0);

GoldendexLauncher.GoldendexLauncher.Frame:TweenSize(oldSize, Enum.EasingDirection.Out,
	Enum.EasingStyle.Quad, 0.3);

wait(0.35);

changeElementState(true);

--------------------------------------------------------------------
-- Get all the launcher data necessary

local newsData = launcherData["NewsData"];
local statusData = launcherData["StatusData"];
local latestVersion = launcherData["LatestVersion"];
local themes = launcherData["Themes"];

--------------------------------------------------------------------
-- Apply the launcher data: News

for i,v in pairs(newsData) do

	local newsOrder = v[1] * -1;	-- This is to make it sort in a different order
	local newsText = v[2];

	if newsText == "@Goldendex-HUB-lua" then
		local div = GoldendexLauncher.NewsDivider:Clone();
		div.LayoutOrder = newsOrder;
		div.Parent = GoldendexLauncher.NewsSection.ContentFrame;
		div.Visible = true;
	else
		local container = GoldendexLauncher.NewsLabel:Clone();
		container.Text = newsText;
		container.LayoutOrder = newsOrder;
		container.Parent = GoldendexLauncher.NewsSection.ContentFrame;
		container.Visible = true;
	end

end

--------------------------------------------------------------------
-- Apply the launcher data: Status

GoldendexLauncher.GoldendexHubStatusFrame.Status.Text = statusData["GoldendexHub"];
GoldendexLauncher.GoldendexAPIStatusFrame.Status.Text = statusData["GoldendexAPI"];
GoldendexLauncher.GoldendexLauncherStatusFrame.Status.Text = statusData["GoldendexLauncher"];

if GoldendexLauncher.GoldendexHubStatusFrame.Status.Text == "Offline" then
	GoldendexLauncher.GoldendexHubStatusFrame.Status.TextColor3 = Color3.fromRGB(192, 49, 52)
end
if GoldendexLauncher.GoldendexAPIStatusFrame.Status.Text == "Offline" then
	GoldendexLauncher.GoldendexAPIStatusFrame.Status.TextColor3 = Color3.fromRGB(192, 49, 52)
end
if GoldendexLauncher.GoldendexLauncherStatusFrame.Status.Text == "Offline" then
	GoldendexLauncher.GoldendexLauncherStatusFrame.Status.TextColor3 = Color3.fromRGB(192, 49, 52)
end

--------------------------------------------------------------------
-- Apply the launcher data: Themes

local selectedTheme = "Default";
local currentThemeIndex = themes[selectedTheme]["ThemeIndex"];
local highestThemeIndex = (function()
	local highest = 0;
	for i,v in pairs(themes) do
		if v["ThemeIndex"] > highest then highest = v["ThemeIndex"]; end
	end
	return highest;
end)();

local function getThemeWithThemeIndex(themeIndex)
	for i,v in pairs(themes) do
		if v["ThemeIndex"] == themeIndex then return i; end
	end
end

local function updateSelectedTheme()
	selectedTheme = getThemeWithThemeIndex(currentThemeIndex);
	GoldendexLauncher.SelectedLabel.Text = selectedTheme;
end

GoldendexLauncher.SelectionRight.MouseButton1Click:Connect(function()
	if currentThemeIndex == highestThemeIndex then
		currentThemeIndex = 1;
	else
		currentThemeIndex = currentThemeIndex + 1;
	end
	updateSelectedTheme();
end)

GoldendexLauncher.SelectionLeft.MouseButton1Click:Connect(function()
	if currentThemeIndex == 1 then
		currentThemeIndex = highestThemeIndex;
	else
		currentThemeIndex = currentThemeIndex - 1;
	end
	updateSelectedTheme();
end)

			GoldendexLauncher.GoldendexLauncher:Destroy();
		end);

-- Dragify

local function dragify(MainFrame)

	local dragging;
	local dragInput;
	local dragStart;
	local startPos;

	local function update(input)
		local Delta = input.Position - dragStart;
		local Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + Delta.X, startPos.Y.Scale, startPos.Y.Offset + Delta.Y);
		game:GetService("TweenService"):Create(MainFrame, TweenInfo.new(.15), {Position = Position}):Play();
	end

	MainFrame.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true;
			dragStart = input.Position;
			startPos = MainFrame.Position;

			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false;
				end
			end)
		end
	end)

	MainFrame.InputChanged:Connect(function(input)
		if (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
			dragInput = input;
		end
	end)

	game:GetService("UserInputService").InputChanged:Connect(function(input)
		if input == dragInput and dragging then
			update(input);
		end
	end)
end

dragify(GoldendexLauncher.GoldendexLauncher.Frame);
